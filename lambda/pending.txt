import json
import boto3
from decimal import Decimal

dynamodb = boto3.resource('dynamodb')
table = dynamodb.Table('notes-table')

# 🔥 FIX: Decimal → float/int
def convert_decimal(obj):
    if isinstance(obj, list):
        return [convert_decimal(i) for i in obj]
    elif isinstance(obj, dict):
        return {k: convert_decimal(v) for k, v in obj.items()}
    elif isinstance(obj, Decimal):
        return float(obj)
    else:
        return obj

def lambda_handler(event, context):
    try:
        print("FULL EVENT:", json.dumps(event))

        # ✅ CLAIMS
        claims = event.get("requestContext", {}) \
                      .get("authorizer", {}) \
                      .get("jwt", {}) \
                      .get("claims", {})

        print("CLAIMS:", claims)

        # ✅ GROUPS FIX
        groups = claims.get("cognito:groups", [])

        if isinstance(groups, str):
            groups = groups.replace("[", "").replace("]", "").replace('"', "").split(",")

        groups = [g.strip() for g in groups if g.strip()]

        print("FINAL GROUPS:", groups)

        # ❌ ADMIN CHECK
        if "admin" not in groups:
            return {
                "statusCode": 403,
                "headers": {"Access-Control-Allow-Origin": "*"},
                "body": json.dumps({"error": "Forbidden"})
            }

        # ✅ FETCH DATA
        response = table.scan()
        items = response.get("Items", [])

        pending = [item for item in items if item.get("status") == "pending"]

        print("PENDING COUNT:", len(pending))

        # 🔥 FIX: convert Decimal
        pending = convert_decimal(pending)

        return {
            "statusCode": 200,
            "headers": {"Access-Control-Allow-Origin": "*"},
            "body": json.dumps(pending)
        }

    except Exception as e:
        print("ERROR:", str(e))
        return {
            "statusCode": 500,
            "headers": {"Access-Control-Allow-Origin": "*"},
            "body": json.dumps({"error": str(e)})
        }
