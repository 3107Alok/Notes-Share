import json
import boto3
from boto3.dynamodb.conditions import Attr

dynamodb = boto3.resource('dynamodb')
table = dynamodb.Table('pyq-table')

def lambda_handler(event, context):
    try:
        subject_code = event.get("queryStringParameters", {}).get("subject_code")
        pyq_type = event.get("queryStringParameters", {}).get("type")

        response = table.scan()

        items = response.get("Items", [])

        # filter manually
        filtered = [
            i for i in items
            if i["subject_code"] == subject_code and i["type"] == pyq_type
        ]

        return {
            "statusCode": 200,
            "headers": {"Access-Control-Allow-Origin": "*"},
            "body": json.dumps(filtered)
        }

    except Exception as e:
        return {
            "statusCode": 500,
            "headers": {"Access-Control-Allow-Origin": "*"},
            "body": json.dumps({"error": str(e)})
        }
