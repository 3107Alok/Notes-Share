import json
import boto3
import base64
import uuid
import PyPDF2
import io
import requests
import os
import re

# AWS
s3 = boto3.client('s3')
dynamodb = boto3.resource('dynamodb')

TABLE_NAME = "syllabus-table"
BUCKET_NAME = "notes-storage-alok123"


# ---------- PDF TEXT ----------
def extract_text_from_pdf(file_bytes):
    try:
        reader = PyPDF2.PdfReader(io.BytesIO(file_bytes))
        text = ""

        for page in reader.pages:
            text += page.extract_text() or ""

        return text.strip()

    except Exception as e:
        print("PDF ERROR:", str(e))
        return ""


# ---------- REGEX FALLBACK ----------
def extract_units_with_regex(text):
    print("🔥 Using REGEX fallback")

    pattern = r"Unit\s*(\d+)\s+([^\n]+)\n(.+?)(?=Unit\s*\d+|$)"
    matches = re.findall(pattern, text, re.DOTALL | re.IGNORECASE)

    units = []
    for m in matches:
        units.append({
            "unit_no": m[0],
            "unit_name": m[1].strip()[:80],
            "syllabus": m[2].strip()[:500]
        })

    return units


# ---------- AI PARSE ----------
def extract_units_smart(text):
    try:
        # 🔥 SUBJECT TRIM
        subject_name = CURRENT_SUBJECT_NAME.lower()
        text_lower = text.lower()

        if subject_name in text_lower:
            idx = text_lower.find(subject_name)
            text = text[idx:]
            print("🔥 Trimmed to subject section")

        # 🔥 CLEAN TEXT
        lines = text.split("\n")
        lines = [l.strip() for l in lines if len(l.strip()) > 20]

        clean_text = "\n".join(lines)

        # 🔥 FULL TEXT (NO MID CUT ❌)
        if len(clean_text) > 6000:
            sample_text = clean_text[:6000]
        else:
            sample_text = clean_text

        print("🔥 SAMPLE LENGTH:", len(sample_text))

        # ---------- PROMPT ----------
        prompt = f"""
Extract ALL units from the syllabus.

For EACH unit extract:
- unit number
- SHORT unit title (max 8 words)
- FULL syllabus

Return ONLY VALID JSON.

[
  {{
    "unit_no": "1",
    "unit_name": "Short title",
    "syllabus": "Full syllabus"
  }}
]

Rules:
- DO NOT skip units
- DO NOT merge units
- unit_name must be short
- max 150 words syllabus
- escape quotes properly

Text:
{sample_text}
"""

        response = requests.post(
            "https://api.groq.com/openai/v1/chat/completions",
            headers={
                "Authorization": f"Bearer {os.environ['GROQ_API_KEY']}",
                "Content-Type": "application/json"
            },
            json={
                "model": "llama-3.1-8b-instant",
                "messages": [{"role": "user", "content": prompt}]
            },
            timeout=10
        )

        data = response.json()

        if "choices" not in data:
            print("GROQ ERROR:", data)
            return []

        output = data["choices"][0]["message"]["content"]
        print("AI RAW:", output)

        # ---------- CLEAN JSON ----------
        output = output.replace("```json", "").replace("```", "").strip()

        start = output.find("[")
        end = output.rfind("]") + 1

        if start != -1 and end != -1:
            output = output[start:end]

        print("🔥 CLEANED:", output)

        # ---------- PARSE ----------
        try:
            return json.loads(output)

        except Exception as e:
            print("FIRST PARSE FAILED:", e)

            try:
                fixed = output

                # 🔥 REMOVE INVALID ESCAPES
                fixed = re.sub(r'\\(?!["\\/bfnrtu])', '', fixed)

                # 🔥 REMOVE NEWLINES
                fixed = fixed.replace("\n", " ")

                # 🔥 FIX TRAILING COMMAS
                fixed = re.sub(r",\s*}", "}", fixed)
                fixed = re.sub(r",\s*]", "]", fixed)

                # 🔥 FIX QUOTES
                fixed = fixed.replace('\\"', '"')

                print("🔥 RETRY:", fixed)

                return json.loads(fixed)

            except Exception as e2:
                print("FINAL PARSE FAILED:", e2)
                return []

    except Exception as e:
        print("AI ERROR:", str(e))
        return []


# ---------- MAIN ----------
def lambda_handler(event, context):
    try:
        print("🔥 FUNCTION STARTED")

        body = event.get("body")

        if isinstance(body, str):
            body = json.loads(body)

        subject_code = body.get("subject_code", "").upper()
        subject_name = body.get("subject_name", "")
        year = str(body.get("year", ""))
        file_data = body.get("file_data")

        if not subject_code or not file_data:
            return {
                "statusCode": 400,
                "headers": {"Access-Control-Allow-Origin": "*"},
                "body": json.dumps({"error": "Missing fields"})
            }

        global CURRENT_SUBJECT_NAME
        CURRENT_SUBJECT_NAME = subject_name

        # ---------- FILE ----------
        file_bytes = base64.b64decode(file_data)

        file_id = str(uuid.uuid4())
        s3_key = f"syllabus/{subject_code}_{file_id}.pdf"

        s3.put_object(
            Bucket=BUCKET_NAME,
            Key=s3_key,
            Body=file_bytes,
            ContentType="application/pdf"
        )

        file_url = f"https://{BUCKET_NAME}.s3.amazonaws.com/{s3_key}"

        # ---------- TEXT ----------
        text = extract_text_from_pdf(file_bytes)

        if not text:
            return {
                "statusCode": 400,
                "headers": {"Access-Control-Allow-Origin": "*"},
                "body": json.dumps({"error": "No text found"})
            }

        print("TEXT LENGTH:", len(text))

        # ---------- AI ----------
        units = extract_units_smart(text)

        # ---------- VALIDATION ----------
        unit_numbers = [str(u.get("unit_no")) for u in units]

        if len(units) < 5 or not all(str(i) in unit_numbers for i in range(1, 6)):
            print("⚠️ Missing units → using regex fallback")
            units = extract_units_with_regex(text)

        # ---------- CLEAN ----------
        clean_units = []

        for u in units:
            name = u.get("unit_name", "")

            if len(name) > 80:
                name = name[:80]

            if not name.strip():
                name = f"Unit {u.get('unit_no')}"

            clean_units.append({
                "unit_no": str(u.get("unit_no", "")),
                "unit_name": name,
                "syllabus": u.get("syllabus") or u.get("content") or ""
            })

        units = clean_units

        print("FINAL UNITS:", units)

        # ---------- DB ----------
        table = dynamodb.Table(TABLE_NAME)

        for unit in units:
            table.put_item(
                Item={
                    "subject_code": subject_code,
                    "unit_no": unit["unit_no"],
                    "unit_name": unit["unit_name"],
                    "syllabus": unit["syllabus"],
                    "subject_name": subject_name,
                    "year": year,
                    "syllabus_url": file_url
                }
            )

        return {
            "statusCode": 200,
            "headers": {"Access-Control-Allow-Origin": "*"},
            "body": json.dumps({
                "message": "Syllabus uploaded",
                "units_saved": len(units)
            })
        }

    except Exception as e:
        print("ERROR:", str(e))
        return {
            "statusCode": 500,
            "headers": {"Access-Control-Allow-Origin": "*"},
            "body": json.dumps({"error": str(e)})
        }
